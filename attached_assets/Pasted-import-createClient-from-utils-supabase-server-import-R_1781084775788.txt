  dateFrom?: string,
  dateTo?: string,
  bookingKey?: string,
) {
  try {
    // Build the query parameters
    const params: Record<string, string> = {}

    if (bookingKey) {
      // If booking key is provided, use it directly

      // Otherwise use product ID and other filters

      params.product_ids = productId

      // Add date range if provided
      if (dateFrom && dateTo) {

      console.error("Error fetching bookings from Regiondo:", error)
      return { data: [] }

      return response.data
    } else {
      return []
    }
  } catch (error) {
    console.error("Error fetching bookings from Regiondo:", error)
    throw error
  }
}

export async function fetchOptionsfromDB(params: {
  productId?: string | number;
  variationId?: string | number;
  Maxdata?: number;
  column?: string;
  dateTime:string;
}): Promise<any> {
  const { productId, variationId, Maxdata, column } = params;

  // Early exits for validation
  if (!supabase) {
    return { success: false, error: "Failed to initialize Supabase client" };
  }
  if (!productId && !variationId) {
    return { success: false, error: "Need ProductId or Variation ID" };
  }

  try {
    // Base query
    let query = supabase.from("product_options").select(column || "*");

    // Apply filters
    if (productId) query = query.eq("product_id", productId);
    if (variationId) query = query.eq("variation_id", variationId);
    if (Maxdata) query = query.limit(Maxdata);

    // Execute query
    const { data: options, error: optionsError } = await query.order("sort_order", { ascending: true });

    if (optionsError) {
      return { success: false, error: optionsError.message };
    }

    // 🟢 Type assertion to ensure TypeScript recognizes it as an object
    if (Maxdata === 1 && column && options.length > 0) {
      // Assert that options[0] is an object
      const firstOption = options[0] as Record<string, any>;

      // Access column safely
      return { success: true, options: firstOption[column as string] };
    }

    return { success: true, options };
  } catch (error) {
    console.error("Error retrieving options for variation:", error);
    return { success: false, error: error instanceof Error ? error.message : "Unknown error" };
  }
}

export async function storeProductVariationsAndOptions(productId: string, variationData: any[]) {
  
  if (!supabase) {
    throw new Error("Failed to initialize Supabase client")
  }

  const results = {
    variations: [] as any[],
    options: [] as any[],
  }

  // Process each variation
  for (const variation of variationData) {
    // Process available dates to only include future dates
    const today = new Date()
    today.setHours(0, 0, 0, 0)

    const availableDates = variation.available_dates
      ? Object.keys(variation.available_dates).filter((date) => new Date(date) >= today)
      : []

    // Prepare variation data for upsert
    const variationRecord = {
      product_id: productId,
      variation_id: variation.variation_id,
      name: variation.name,
      from_date: variation.from,
      to_date: variation.to,
      appointment_type: variation.appointment_type,
      available_dates: availableDates,
      updated_at: new Date().toISOString(),
    }

    // Upsert variation data
    const { data: upsertedVariation, error: variationError } = await supabase
      .from("product_variations")
      .upsert(variationRecord, { onConflict: "product_id, variation_id" })

    if (variationError) {
      console.error(`Error upserting variation ${variation.variation_id}:`, variationError)
      continue
    }

    results.variations.push(upsertedVariation)

    // Fetch available options for this variation
    const today_str = new Date().toISOString().split("T")[0] // Format: YYYY-MM-DD
    try {
      const options = await fetchAvailableOptions(variation.variation_id, today_str)
      if (options && Object.keys(options).length > 0) {
        // Delete existing options for this product and variation
        const { error: deleteError } = await supabase
          .from("product_options")
          .delete()
          .eq("product_id", productId)
          .eq("variation_id", variation.variation_id)

        if (deleteError) {
          console.error(`Error deleting existing options for variation ${variation.variation_id}:`, deleteError)
          continue
        }
        // console.log("Object.values(options)")
        // console.log(Object.values(options))
        // Insert new options
        const optionRecords = Object.values(options).map((option: any) => {
          if(!option.option_id) {
            return false
          }
            return {
            product_id: productId,
            variation_id: variation.variation_id,
            option_id: option.option_id,
            name: option.name,
            sort_order: option.sort_order,
            min_qty_to_sell: option.min_qty_to_sell,
            max_qty_to_sell: option.max_qty_to_sell,
            original_price: option.original_price,
            regiondo_price: option.regiondo_price,
            vat_percentage_val: option.vat_percentage_val,
            duration_value: option.duration_value,
            duration_type: option.duration_type,
            description: option.description,
            sales_start_dt: option.sales_start_dt,
            sales_end_dt: option.sales_end_dt,
            capacity: option.capacity,
            booking_notice_period: option.booking_notice_period,
            qty_left: option.qty_left,
            created_at: new Date().toISOString(),
            updated_at: new Date().toISOString(),
          }
        })

        if (optionRecords.length > 0) {
          const { data: insertedOptions, error: insertError } = await supabase
            .from("product_options")
            .insert(optionRecords)
            .select()

          if (insertError) {
            console.error(`Error inserting options for variation ${variation.variation_id}:`, insertError)
            continue
          }

          results.options.push(...(insertedOptions || []))
        }
      }
    } catch (error) {
      console.error(`Error fetching options for variation ${variation.variation_id}:`, error)
    }
  }

  return results
}

/**
 * Helper function to format a date as YYYY-MM-DD
 * @param date - The date to format
 * @returns Formatted date string
 */
export function formatDateYMD(date: Date): string {
  return date.toISOString().split("T")[0]
}

/**
 * Helper function to get a date range for the next N days
 * @param days - Number of days to include in the range
 * @returns Object with fromDate and toDate strings
 */
export function getDateRangeForNextDays(days = 30) {
  const today = new Date()
  const endDate = new Date()
  endDate.setDate(today.getDate() + days)

  return {
    fromDate: formatDateYMD(today),
    toDate: formatDateYMD(endDate),
  }
}

  variations: VariationWithOptions[]
}

/**
 * Fetch complete product details with all variations, dates, timeslots, and available options
 * @param productId - The ID of the product
 * @returns Promise with the complete product details
 */
export async function fetchCompleteProductDetailsWithOptions(
  productId: string | number,
): Promise<CompleteProductDetails> {
  try {
    console.log(`Fetching complete product details with options for product ${productId}...`)

    // Step 1: Fetch the product details
    const product = await fetchProductById(productId)
    if (!product || !product.data) {
      throw new Error(`Product ${productId} not found or invalid response`)
    }

    // Step 2: Extract variations from the product data
    const productVariations = product.data.variations || []
    if (!productVariations.length) {
      throw new Error(`No variations found for product ${productId}`)
    }

    // Step 3: Process each variation to get options for each date and time
    const variationsWithOptions: VariationWithOptions[] = []

    for (const variation of productVariations) {
      const variationId = variation.variation_id

      // Initialize the variation with options structure
      const variationWithOptions: VariationWithOptions = {
        variation_id: variationId,
        name: variation.name,
        from: variation.from,
        to: variation.to,
        appointment_type: variation.appointment_type,
        dates: {},
      }

      // Process each date in the variation
      if (variation.available_dates) {
        for (const [date, timesArrays] of Object.entries(variation.available_dates)) {
          // Initialize the date structure
          variationWithOptions.dates[date] = { times: {} }

          // Process each time array for the date
          for (const timeArray of timesArrays as string[][]) {
            for (const time of timeArray) {
              // Initialize the time structure
              variationWithOptions.dates[date].times[time] = { options: {} }

              try {
                // Fetch available options for this variation and date
                const options = await fetchAvailableOptions(variationId, date)

                // Process each option
                if (options && typeof options === "object") {
                  for (const [optionId, optionData] of Object.entries(options)) {
                    // Add the option to the structure
                    variationWithOptions.dates[date].times[time].options[optionId] = {
                      ...optionData,
                      option_id: optionId,
                    }
                  }
                }
              } catch (error) {
                console.error(`Error fetching options for variation ${variationId} on ${date} at ${time}:`, error)
                // Continue with the next time slot even if this one fails
              }
            }
          }
        }
      }

      // Add the processed variation to the result
      variationsWithOptions.push(variationWithOptions)
    }

    return {
      product,
      variations: variationsWithOptions,
    }
  } catch (error) {
    console.error(`Error fetching complete product details with options for product ${productId}:`, error)
    throw error
  }
}

/**
 * Get available capacity for a specific variation, date, and time
 * @param productDetails - The complete product details
 * @param variationId - The variation ID
 * @param date - The date in format YYYY-MM-DD
 * @param time - The time in format HH:MM:SS
 * @returns Object with capacity information
 */
export function getAvailableCapacity(
  productDetails: CompleteProductDetails,
  variationId: string,
  date: string,
  time: string,
): {
  totalCapacity: number
  availableCapacity: number
  options: Array<{
    optionId: string
    name: string
    capacity: number
    available: number
    price: string
  }>
} {
  // Find the variation
  const variation = productDetails.variations.find((v) => v.variation_id === variationId)
  if (!variation || !variation.dates[date] || !variation.dates[date].times[time]) {
    return { totalCapacity: 0, availableCapacity: 0, options: [] }
  }

  // Get the options for this variation, date, and time
  const options = variation.dates[date].times[time].options

  // Calculate total capacity and available capacity
  let totalCapacity = 0
  let availableCapacity = 0
  const optionsArray = []

  for (const [optionId, optionData] of Object.entries(options)) {
    const capacity = Number(optionData.capacity) || 0
    const available = Number(optionData.qty_left) || 0

    totalCapacity += capacity
    availableCapacity += available

    optionsArray.push({
      optionId,
      name: optionData.name,
      capacity,
      available,
      price: optionData.regiondo_price || optionData.original_price,
    })
  }

  return {
    totalCapacity,
    availableCapacity,
    options: optionsArray,
  }
}

/**
 * Calculate resource usage based on timeslot data
 * @param timeslotData - The timeslot data from the API
 * @returns Resource usage percentage
 */
export function calculateResourceUsage(timeslotData: any): number {
  if (!timeslotData || !timeslotData.event_capacity || timeslotData.event_capacity === 0) {
    return 0
  }

  const booked = timeslotData.event_capacity - timeslotData.qty_available
  return Math.round((booked / timeslotData.event_capacity) * 100)
}

/**
 * Fetch sold items for a specific product ID
 * @param productId - The ID of the product to fetch sold items for
 * @param dateFrom - Optional start date in format YYYY-MM-DD
 * @param dateTo - Optional end date in format YYYY-MM-DD
 * @returns Promise with the list of sold items
 */
export async function fetchSoldItemsByProductId(
  productId: string | number,
  dateFrom?: string,
  dateTo?: string,
): Promise<SoldItem[]> {
  try {
    const params: Record<string, string> = {
      product_ids: productId.toString(),
    }

    // Add date range if provided
    if (dateFrom && dateTo) {
      params.date_range = `${dateFrom},${dateTo}`
      params.date_bought = "Date of event" // Use event date instead of purchase date
    }

    const response = await regiondoApi.get("/partner/solditems", params)

    if (!response || !Array.isArray(response)) {
      console.warn("Unexpected response format for sold items:", response)
      return []
    }

    console.log(`Found ${response.length} sold items for product ${productId}`)
    return response as SoldItem[]
  } catch (error) {
    console.error(`Error fetching sold items for product ${productId}:`, error)
    throw error
  }
}

/**
 * Get sold items grouped by date and variation
 * @param soldItems - Array of sold items from the API
 * @returns Object with dates as keys and arrays of items grouped by variation as values
 */
export function groupSoldItemsByDateAndVariation(soldItems: SoldItem[]): Record<string, Record<string, SoldItem[]>> {
  const grouped: Record<string, Record<string, SoldItem[]>> = {}

  soldItems.forEach((item) => {
    // Use event_date as the key
    const date = item.event_date
    const variationId = item.option_id.toString()

    // Initialize date group if it doesn't exist
    if (!grouped[date]) {
      grouped[date] = {}
    }

    // Initialize variation group if it doesn't exist
    if (!grouped[date][variationId]) {
      grouped[date][variationId] = []
    }

    // Add item to the appropriate group
    grouped[date][variationId].push(item)
  })

  return grouped
}

/**
 * Calculate booking statistics from sold items
 * @param soldItems - Array of sold items from the API
 * @returns Object with booking statistics
 */
export function calculateBookingStats(soldItems: SoldItem[]): {
  totalBookings: number
  totalRevenue: number
  currencyCode: string
  bookingsByVariation: Record<string, number>
  revenueByVariation: Record<string, number>
} {
  let totalBookings = 0
  let totalRevenue = 0
  const currencyCode = soldItems.length > 0 ? soldItems[0].price_incl_tax.currency_code : "EUR"
  const bookingsByVariation: Record<string, number> = {}
  const revenueByVariation: Record<string, number> = {}

  soldItems.forEach((item) => {
    // Skip cancelled items
    if (item.cancel_date) {
      return
    }

    totalBookings++
    totalRevenue += item.price_incl_tax.amount

    const variationName = item.variation_name || "Unknown"

    // Count bookings by variation
    if (!bookingsByVariation[variationName]) {
      bookingsByVariation[variationName] = 0
    }
    bookingsByVariation[variationName]++

    // Sum revenue by variation
    if (!revenueByVariation[variationName]) {
      revenueByVariation[variationName] = 0
    }
    revenueByVariation[variationName] += item.price_incl_tax.amount
  })

  return {
    totalBookings,
    totalRevenue,
    currencyCode,
    bookingsByVariation,
    revenueByVariation,
  }
}

/**
 * Create a permanent reservation in Regiondo
 * @param data - The reservation data
 * @param placeBooking - Whether to place the booking immediately (default: true)
 * @returns Promise with the reservation response and purchase response
 */
export async function createPermanentReservation(
  data: PermanentReservationRequest,
  placeBooking = true,
): Promise<{ reservation: PermanentReservationResponse; purchase?: PurchaseResponse; error?: string }> {
  try {
    console.log(`Creating permanent reservation for product ${data.product_id}`)

    // Format the request data
    const requestData = {
      ...data,
      skip_customer_validation: data.skip_customer_validation ?? true,
    }

    console.log("Creating reservation with data:", requestData)

    // Step 1: Create the permanent reservation
    const reservationResponse = await regiondoApi.post(
      `/checkout/permanentreservation?place_booking=${placeBooking}`,
      requestData,
    )

    // Check if booking_key is in the response
    if (!reservationResponse || !reservationResponse.booking_key) {
      const errorMessage =
        reservationResponse?.error || "Failed to create permanent reservation: No booking key received"
      console.error(errorMessage, reservationResponse)
      return {
        reservation: reservationResponse || {},
        error: errorMessage,
      }
    }

    console.log(`Successfully created permanent reservation with booking key: ${reservationResponse.booking_key}`)

    // Step 2: Complete the purchase using the booking key
    try {
      const bookingKey = reservationResponse.booking_key
      console.log(`Completing purchase for booking key: ${bookingKey}`)

      const purchaseResponse = await regiondoApi.post(
        `/checkout/permanentreservation/purchase?booking_key=${bookingKey}`,
      )

      if (!purchaseResponse || !purchaseResponse.order_id) {
        const errorMessage = purchaseResponse?.error || "Failed to complete purchase: No order ID received"
        console.error(errorMessage, purchaseResponse)
        return {
          reservation: reservationResponse,
          error: errorMessage,
        }
      }

      console.log(`Successfully completed purchase with order ID: ${purchaseResponse.order_id}`)

      // Step 3: Fetch the booking details using the booking key and store in database
      try {
        console.log(`Fetching booking details for booking key: ${bookingKey}`)

        // Fetch booking details
        const bookingDetails = await fetchBookingsByProductId(
          data.product_id.toString(),
          undefined,
          undefined,
          bookingKey,
        )

        if (bookingDetails && bookingDetails.length > 0) {
          console.log(`Successfully fetched booking details for booking key: ${bookingKey}`)

          // Store the booking in the database
          const storeResponse = await fetch(`/api/bookings/${data.product_id}/store`, {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
            },
            body: JSON.stringify({ bookings: bookingDetails }),
          })

          if (!storeResponse.ok) {
            console.error(`Failed to store booking in database: ${await storeResponse.text()}`)
          } else {
            console.log(`Successfully stored booking in database`)
          }
        } else {
          console.warn(`No booking details found for booking key: ${bookingKey}`)
        }
      } catch (fetchError) {
        console.error(`Error fetching or storing booking details: ${fetchError.message}`)
        // Continue with the purchase response even if fetching/storing booking fails
      }

      return {
        reservation: reservationResponse,
        purchase: purchaseResponse,
      }
    } catch (purchaseError) {
      console.error("Error completing purchase:", purchaseError)
      return {
        reservation: reservationResponse,
        error: `Reservation created but purchase failed: ${purchaseError.message || "Unknown error"}`,
      }
    }
  } catch (error) {
    console.error("Error creating permanent reservation:", error)
    return {
      reservation: {} as PermanentReservationResponse,
      error: `Failed to create reservation: ${error.message || "Unknown error"}`,
    }
  }
}
